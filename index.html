// A&J Cleaning Services GmbH - Google Sheets API // Stundenzettel + Dienstplan + Tagesabrechnung function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var type = data.type || ''; if (type === 'stundenzettel') { saveStundenzettel(ss, data); } else if (type === 'tagesabrechnung') { saveTagesabrechnung(ss, data); } else if (type === 'dienstplan') { saveDienstplan(ss, data); } else { saveAllgemein(ss, data); } return ContentService .createTextOutput(JSON.stringify({status:'ok', hotel:data.hotel, type:type})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error', message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput('A&J Manager API OK | Bereit') .setMimeType(ContentService.MimeType.TEXT); } function getOrCreateSheet(ss, name) { var ws = ss.getSheetByName(name); if (!ws) { ws = ss.insertSheet(name); } return ws; } function styleHeader(ws, cols) { var hdr = ws.getRange(1, 1, 1, cols); hdr.setBackground('#0d1b2a'); hdr.setFontColor('#c8a84b'); hdr.setFontWeight('bold'); ws.setFrozenRows(1); } function saveStundenzettel(ss, data) { var sheetName = 'SZ - ' + (data.hotel || 'Unbekannt'); var ws = getOrCreateSheet(ss, sheetName); if (ws.getLastRow() === 0) { var hdrs = ['Zeitstempel','Hotel','Vorarbeiter','Mitarbeiter','Monat','Ist-Std','Zimmer','Grundrein.']; ws.getRange(1, 1, 1, hdrs.length).setValues([hdrs]); styleHeader(ws, hdrs.length); } var row = [ new Date().toLocaleString('de-DE'), data.hotel || '', data.vorarbeiter || '', data.ma || '', data.mon || '', data.istGes || '', data.ziGes || '', data.grGes || '' ]; ws.appendRow(row); } function saveTagesabrechnung(ss, data) { var sheetName = 'TA - ' + (data.hotel || 'Unbekannt'); var ws = getOrCreateSheet(ss, sheetName); if (ws.getLastRow() === 0) { var hdrs = ['Zeitstempel','Hotel','Vorarbeiter','Datum','Mitarbeiter','Zimmer','Grundrein.','Check','Sonder-Std','Ist-Std']; ws.getRange(1, 1, 1, hdrs.length).setValues([hdrs]); styleHeader(ws, hdrs.length); } var rows = data.rows || []; var datum = data.datum || ''; var ts = new Date().toLocaleString('de-DE'); rows.forEach(function(r) { if (!r.name && !r.zi) return; ws.appendRow([ ts, data.hotel || '', data.vorarbeiter || '', datum, r.name || '', r.zi || '', r.gr || '', r.chk || '', r.son || '', r.ist || '' ]); }); } function saveDienstplan(ss, data) { var sheetName = 'DP - ' + (data.hotel || 'Unbekannt'); var ws = getOrCreateSheet(ss, sheetName); if (ws.getLastRow() === 0) { var hdrs = ['Zeitstempel','Hotel','Vorarbeiter','Von','Bis','Mitarbeiter','Mo','Di','Mi','Do','Fr','Sa','So']; ws.getRange(1, 1, 1, hdrs.length).setValues([hdrs]); styleHeader(ws, hdrs.length); } var dp = data.dp || {}; var names = dp.names || []; var shifts = dp.shifts || []; var ts = new Date().toLocaleString('de-DE'); names.forEach(function(name, i) { if (!name) return; var row = [ts, data.hotel || '', data.vorarbeiter || '', dp.von || '', dp.bis || '', name]; var dayShifts = shifts[i] || []; for (var d = 0; d < 7; d++) { row.push(dayShifts[d] || ''); } ws.appendRow(row); }); } function saveAllgemein(ss, data) { var ws = getOrCreateSheet(ss, 'Alle Daten'); if (ws.getLastRow() === 0) { ws.getRange(1,1,1,3).setValues([['Zeitstempel','Hotel','Daten']]); styleHeader(ws, 3); } ws.appendRow([new Date().toLocaleString('de-DE'), data.hotel || '', JSON.stringify(data)]); }
